pipeline{
    agent any
    environment {
    FIREBASE_DEPLOY_TOKEN = credentials('firebase-token')
    }

    stages{
        stage('Building'){
            steps{
	    sh 'npm install -g firebase-tools'
	    sh 'npm install -g npm'
            sh 'docker -v'
            }
        } 

         stage('Testing'){
            steps{
            sh'npm -v'
            }
        }

        stage('Staging'){
            steps{
            sh 'firebase help'
            }
        }

        stage('Production'){
            steps{
            sh 'firebase deploy -P jenkins-aws-production --token "$FIREBASE_DEPLOY_TOKEN"'
            }
        }
    }
}
